% ---------- STYLES ----------
\tikzset{
	basic box/.style = {
		shape = rectangle,
		draw = #1,
		fill = #1!20,
		rounded corners,
		align = center,
		text width=6cm,      
		minimum height=2.5cm, 
		inner sep=4pt             
	},
	header node/.style = {
		minimum width = 2cm,
		font = \strut\Large\ttfamily,
		text depth = 1pt,
		fill = white,
		draw,
		rounded corners=2pt,
		fill = yellow!40
	},
	future node/.style = {
		circle,
		draw=futuregreen,
		fill=futuregreen!25,
		thick,
		minimum size=2.2cm,
		font=\bfseries\Large
	},
	header/.style = {
		inner ysep = 1.1em,
		append after command = {
			\pgfextra{\let\TikZlastnode\tikzlastnode}
			node[header node] (header-\TikZlastnode) at (\TikZlastnode.north) {#1}
			node[fit=(\TikZlastnode)(header-\TikZlastnode)] (h-\TikZlastnode) {}
		}
	}
}
